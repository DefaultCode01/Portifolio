🗄️ Principais Comandos SQL

Comando Descrição Exemplo
CREATE Cria tabela ou banco de dados CREATE TABLE usuarios(id INT, nome VARCHAR(50));
INSERT Insere dados na tabela INSERT INTO usuarios VALUES(1, 'Matheus');
SELECT Consulta dados da tabela SELECT * FROM usuarios;
UPDATE Atualiza dados existentes UPDATE usuarios SET nome='Lucas' WHERE id=1;
DELETE Remove dados da tabela DELETE FROM usuarios WHERE id=1;
JOIN Relaciona tabelas SELECT * FROM pedidos JOIN usuarios ON pedidos.id_usuario=usuarios.id;
